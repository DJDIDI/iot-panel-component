## Switch

```jsx
import React, {useState} from 'react';
import { Switch } from 'qcloud-iot-panel-component';


export default () => (
  <div style={{width: 200}}>
    <h2>基础组件Icon</h2>
    <Switch disabled />
    <Switch />
    <Switch disabled checked />
    <Switch checked />
  </div>
)
```
